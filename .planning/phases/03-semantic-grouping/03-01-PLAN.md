   - Update `Message::DiffParsed` handler in `update()`:
     After `self.apply_new_diff_data(new_data)`:
     ```rust
     if self.claude_available {
         // Cancel in-flight grouping (ROB-05)
         if let Some(handle) = self.grouping_handle.take() {
             handle.abort();
         }
         self.grouping_status = GroupingStatus::Loading;
         let summaries = crate::grouper::file_summaries(&self.diff_data);
         Some(Command::SpawnGrouping(summaries))
     } else {
         self.grouping_status = GroupingStatus::Idle;
         None
     }
     ```
     Note: this replaces the current `None` return from DiffParsed handler.
   - Add handler for `Message::GroupingComplete(groups)`:
     ```rust
     self.semantic_groups = Some(groups);
     self.grouping_status = GroupingStatus::Done;
     self.grouping_handle = None;
     None
     ```
   - Add handler for `Message::GroupingFailed(err)`:
     ```rust
     tracing::warn!("Grouping failed: {}", err);
     self.grouping_status = GroupingStatus::Error(err);
     self.grouping_handle = None;
     None  // Continue showing ungrouped -- graceful degradation (ROB-06)
     ```

3. In `src/main.rs`, add `SpawnGrouping` to the command executor:
   ```rust
   Command::SpawnGrouping(summaries) => {
       let tx2 = tx.clone();
       let handle = tokio::spawn(async move {
           match crate::grouper::llm::request_grouping_with_timeout(&summaries).await {
               Ok(groups) => {
                   let _ = tx2.send(Message::GroupingComplete(groups)).await;
               }
               Err(e) => {
                   let _ = tx2.send(Message::GroupingFailed(e.to_string())).await;
               }
           }
       });
       app.grouping_handle = Some(handle);
   }
   ```
