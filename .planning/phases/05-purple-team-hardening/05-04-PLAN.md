```rust
use tokio::io::AsyncReadExt;

// Instead of: let output = child.wait_with_output().await?;
// Use bounded read:
let stdout = child.stdout.take()
    .ok_or_else(|| anyhow::anyhow!("failed to capture stdout"))?;
let mut reader = tokio::io::BufReader::new(stdout);
let mut buf = Vec::with_capacity(8192);
let bytes_read = reader.take(MAX_RESPONSE_BYTES as u64)
    .read_to_end(&mut buf).await?;

if bytes_read >= MAX_RESPONSE_BYTES {
    // Kill the child process and return error
    child.kill().await.ok();
    anyhow::bail!("LLM response exceeded {} byte limit", MAX_RESPONSE_BYTES);
}

let status = child.wait().await?;
if !status.success() {
    anyhow::bail!("LLM exited with status {}", status);
}
let stdout_str = String::from_utf8(buf)?;
```

fn truncate(s: &str, max: usize) -> &str {

Replacement (safe):
```rust
/// Truncate a string to at most `max` bytes, respecting UTF-8 char boundaries.
/// Returns a string slice that is always valid UTF-8.
fn truncate(s: &str, max: usize) -> &str {
    if s.len() <= max {
        s
    } else {
        // Find the largest char boundary <= max
        let mut end = max;
        while end > 0 && !s.is_char_boundary(end) {
            end -= 1;
        }
        &s[..end]
    }
}
```

This uses `is_char_boundary()` which is stable since Rust 1.9. The function walks backward from `max` to find the nearest valid UTF-8 boundary. This is O(1) since UTF-8 chars are at most 4 bytes.
